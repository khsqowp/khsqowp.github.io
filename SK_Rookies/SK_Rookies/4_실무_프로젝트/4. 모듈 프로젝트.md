## 2. 🧪 프로젝트 구현 가이드

### 2.1. 🧹 데이터 준비: `network_traffic.csv` 생성

먼저 프로젝트에 사용할 가상의 네트워크 트래픽 로그 파일을 생성합니다.

```python
import pandas as pd
import numpy as np
from datetime import datetime, timedelta

# 가상 데이터 생성 함수
def generate_traffic_log(num_entries=1000):
    ips = [f"192.168.1.{i}" for i in range(1, 10)] + \
          [f"10.0.0.{i}" for i in range(1, 5)] + \
          ["203.0.113.10", "185.191.205.10"] # 악성 의심 IP 추가
    
    ports = [22, 80, 443, 3389, 53, 21, 23, 8080]
    actions = ['ALLOW', 'DENY']
    
    start_time = datetime(2023, 11, 12, 9, 0, 0)
    
    data = []
    for i in range(num_entries):
        timestamp = start_time + timedelta(seconds=np.random.randint(0, 3600 * 24)) # 24시간 내 랜덤
        source_ip = np.random.choice(ips, p=[0.1]*9 + [0.05]*4 + [0.02, 0.03]) # 특정 IP 확률 높임
        dest_port = np.random.choice(ports, p=[0.2, 0.3, 0.3, 0.1, 0.05, 0.02, 0.01, 0.02])
        action = np.random.choice(actions, p=[0.9, 0.1])
        
        # 악성 의심 IP에 대한 DENY 액션 확률 높임
        if source_ip in ["203.0.113.10", "185.191.205.10"]:
            action = np.random.choice(['DENY', 'ALLOW'], p=[0.8, 0.2])
            dest_port = np.random.choice([22, 3389, 23], p=[0.4, 0.4, 0.2]) # 특정 포트 집중
        
        data.append([timestamp, source_ip, dest_port, action])
        
    df = pd.DataFrame(data, columns=['timestamp', 'source_ip', 'dest_port', 'action'])
    df = df.sort_values(by='timestamp').reset_index(drop=True)
    return df

# 로그 파일 생성
traffic_df = generate_traffic_log(num_entries=5000)
traffic_df.to_csv('network_traffic.csv', index=False)
print("✅ 'network_traffic.csv' 파일이 생성되었습니다.")
print(traffic_df.head())
```

#### **단계 3: IP 정보 조회 (API 활용)**

```python
import requests
import json
import time

def get_ip_info(ip):
    """ip-api.com을 통해 IP 정보를 조회하는 함수"""
    try:
        response = requests.get(f"http://ip-api.com/json/{ip}")
        response.raise_for_status()
        return response.json()
    except requests.RequestException as e:
        print(f"❌ IP 정보 조회 실패 ({ip}): {e}")
        return None

ip_details = {}
if malicious_ips:
    print("\n--- 📡 IP 정보 조회 중 ---")
    for ip in malicious_ips:
        info = get_ip_info(ip)
        if info and info.get('status') == 'success':
            ip_details[ip] = {
                'country': info.get('country', 'N/A'),
                'city': info.get('city', 'N/A'),
                'isp': info.get('isp', 'N/A')
            }
            print(f"  - {ip}: {ip_details[ip]['country']} ({ip_details[ip]['isp']})")
        time.sleep(1) # API Rate Limit 준수
```

#### **단계 4: Slack 알림 전송 (API 활용)**

```python
import os

def send_slack_alert(message):
    """Slack Webhook을 통해 알림을 보내는 함수"""
    # ⚠️ 실제 Slack Webhook URL은 환경 변수 등으로 관리해야 합니다.
    #    (예: export SLACK_WEBHOOK_URL="your_real_url")
    webhook_url = os.environ.get("SLACK_WEBHOOK_URL")
    if not webhook_url:
        print("❌ Slack 알림 전송 실패: SLACK_WEBHOOK_URL 환경 변수가 설정되지 않았습니다.")
        return

    headers = {'Content-type': 'application/json'}
    payload = {'text': message}
    try:
        response = requests.post(webhook_url, headers=headers, data=json.dumps(payload))
        response.raise_for_status()
        print("✅ Slack 알림 전송 성공!")
    except requests.RequestException as e:
        print(f"❌ Slack 알림 전송 실패: {e}")

if malicious_ips:
    alert_message = "🚨 *보안 경고: 악성 IP 탐지 보고서* 🚨\n"
    alert_message += "--------------------------------------\n"
    for ip in malicious_ips:
        details = ip_details.get(ip, {})
        alert_message += f"- *IP*: `{ip}`\n"
        alert_message += f"  - *DENY 횟수*: {ip_deny_counts[ip]}회\n"
        alert_message += f"  - *국가*: {details.get('country', 'N/A')}\n"
        alert_message += f"  - *ISP*: {details.get('isp', 'N/A')}\n"
        alert_message += "--------------------------------------\n"
    alert_message += "자세한 내용은 보안 대시보드를 확인해주세요."
    
    send_slack_alert(alert_message)
else:
    print("\n✅ 탐지된 악성 IP가 없어 Slack 알림을 보내지 않습니다.")
```

        print("❌ Slack 알림 전송 실패: SLACK_WEBHOOK_URL 환경 변수가 설정되지 않았습니다.")
        return

    headers = {'Content-type': 'application/json'}
    payload = {'text': message}
    try:
        response = requests.post(webhook_url, headers=headers, data=json.dumps(payload))
        response.raise_for_status()
        print("✅ Slack 알림 전송 성공!")
    except requests.RequestException as e:
        print(f"❌ Slack 알림 전송 실패: {e}")

if vt_malicious_ips:
    alert_message = "🚨 *침해사고 경고: 악성 IP 자동 차단 조치 완료* 🚨\n"
    alert_message += "--------------------------------------\n"
    for ip in vt_malicious_ips:
        alert_message += f"- *차단된 IP*: `{ip}`\n"
        alert_message += f"  - *탐지 근거*: 네트워크 로그 비정상 DENY 시도 및 VirusTotal 악성 확인\n"
        alert_message += "--------------------------------------\n"
    alert_message += "추가 분석 및 대응이 필요합니다."
    send_slack_alert(alert_message)
elif malicious_ips_detected:
    alert_message = "⚠️ *보안 알림: 의심 IP 탐지 (VirusTotal 미확인)* ⚠️\n"
    alert_message += "--------------------------------------\n"
    for ip in malicious_ips_detected:
        alert_message += f"- *의심 IP*: `{ip}`\n"
        alert_message += f"  - *탐지 근거*: 네트워크 로그 비정상 DENY 시도\n"
        alert_message += "--------------------------------------\n"
    alert_message += "수동 확인 및 추가 분석이 필요합니다."
    send_slack_alert(alert_message)
else:
    print("\n✅ 특이사항 없어 Slack 알림을 보내지 않습니다.")

# --- 5. 보고서 생성 (시각화) ---
if malicious_ips_detected:
    plt.figure(figsize=(10, 6))
    sns.barplot(x=ip_deny_counts.index, y=ip_deny_counts.values, palette='coolwarm')
    plt.title('📊 탐지된 악성 의심 IP별 DENY 시도 횟수', fontsize=16)
    plt.xlabel('Source IP', fontsize=12)
    plt.ylabel('DENY 시도 횟수', fontsize=12)
    plt.xticks(rotation=45, ha='right')
    plt.tight_layout()
    plt.show()
else:
    print("\n📊 시각화할 데이터가 없습니다.")
